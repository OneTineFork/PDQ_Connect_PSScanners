$dism = (dism.exe /Online /Cleanup-Image /CheckHealth) 2>&1 | Out-String
$dismTrim = $dism.Trim()

[pscustomobject]@{
  Check = "DISM_CheckHealth"
  Status = if ($dismTrim -match "No component store corruption detected") { "OK" }
           elseif ($dismTrim -match "component store is repairable") { "WARN" }
           elseif ($dismTrim -match "component store corruption detected") { "FAIL" }
           else { "UNKNOWN" }
  Detail = ($dismTrim -replace '\s+', ' ').Substring(0, [Math]::Min(250, ($dismTrim -replace '\s+', ' ').Length))
}
